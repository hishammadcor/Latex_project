	}
}

%	general table styles
%--------------------------------------------------------------------------------------------------------%
\pgfplotstableset{
	col sep                     = semicolon,
    row sep                     = newline,
    debug						= false,																		% if enabled, prints table code into log
    string type,																								% cells contain strings/mixed data
    percent is letter			= true,																			% percent is no comment within this table
	trim cells					= true,																			% trim leading/trailing whitespaces
	empty cells with			= {},																			% replace empty cells with empty string
%
    begin table					= {
									 \setmainfont{Segoe UI}
									 \selectfont
									 \footnotesize																% set a font size by pt does not work...
									 \selectfont
                                    \setstretch{1.0}
                                    \begin{tblr}																% use tblr to have alignment, width and coloring accessible
                                  },
    end table					= {
                                    \end{tblr}
                                    \normalspacing
                                    \vspace{6pt}																% 28pt spacing after table
                                    \newline
                                  },
%
	header                      = false,																		% both lines needed to get rid of indices in empty header cells
	every head row/.style       = { output empty row },
%
    highlight_italics/.append style={																			% highlight first column in italics
        postproc cell content/.append code={
                \pgfkeysalso{@cell content=\setmainfont{Segoe UI}\footnotesize\selectfont\textit{##1}}%
        },
    },
    highlight_italics first row/.style={																		% highlight first row in italics
        postproc cell content/.append code={
            \count0=\pgfplotstablerow
            \advance\count0 by1
            \ifnum\count0=1
                \pgfkeysalso{@cell content=\setmainfont{Segoe UI}\footnotesize\selectfont\textit{##1}}%
            \fi
        },
    },
%
    highlight_bold/.append style={																				% highlight first column in bold
        postproc cell content/.append code={
                \pgfkeysalso{@cell content=\setmainfont{Segoe UI}\footnotesize\selectfont\textbf{##1}}%
        },
    },
    highlight_bold first row/.style={																			% highlight first row in bold
        postproc cell content/.append code={
            \count0=\pgfplotstablerow
            \advance\count0 by1
            \ifnum\count0=1
                \pgfkeysalso{@cell content=\setmainfont{Segoe UI}\footnotesize\selectfont\textbf{##1}}%
            \fi
        },
    },
%
    highlight_bold_rotate first row/.style={																	% rotate first row by 90° and limit height to break CLDH into two lines as small as possible
        postproc cell content/.append code={
            \count0=\pgfplotstablerow
            \advance\count0 by1
            \ifnum\count0=1
                \pgfkeysalso{@cell content=\setmainfont{Segoe UI}\footnotesize\selectfont\rotatebox{90}{\parbox{3.2cm}{\textbf{##1}}}}%
            \fi
        },
    },
%
    highlight_bold_rotate_studienfaelle first row/.style={
        postproc cell content/.append code={
            \count0=\pgfplotstablerow
            \advance\count0 by1
            \ifnum\count0=1
                \pgfkeysalso{@cell content=\setmainfont{Segoe UI}\footnotesize\selectfont{\textit{##1}}}%
            \fi
        },
    },
%
    highlight_bold_rotate_studienverlauf first row/.style={
        postproc cell content/.append code={
            \count0=\pgfplotstablerow
            \advance\count0 by1
            \ifnum\count0=1
                \pgfkeysalso{@cell content=\setmainfont{Segoe UI}\footnotesize\selectfont{\textit{##1}}}%
            \fi
        },
    },
}
